local success, response = pcall(function()
    return HttpService:GetAsync("https://api.example.com/playerData/" .. player.UserId)
end)

if not success then
    console.log("Error fetching data for " .. player.Name .. ": " .. response)
    return nil
end

local data = HttpService:JSONDecode(response)
assert(data.score, "Score must exist in player data")
assert(data.health, "Health must exist in player data")

return data

local function updatePlayerStats(player, data)
    assert(player, "Player must not be nil")
    assert(data, "Data must not be nil")
    
    if data then
        player:SetAttribute("Score", data.score)
        player:SetAttribute("Health", data.health)
        DebugModule.log(player.Name .. " stats updated: Score = " .. data.score .. ", Health = " .. data.health)
    end
end

local function onPlayerAdded(player)
    local playerData = fetchPlayerData(player)
    if playerData then
        updatePlayerStats(player, playerData)
    end
end

local function setupGame()
    for _, player in ipairs(Players:GetPlayers()) do
        onPlayerAdded(player)
    end
    
    Players.PlayerAdded:Connect(onPlayerAdded)
end

local function main()
    setupGame()
end
